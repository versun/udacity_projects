import random
import numpy as np

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    rd = np.random.choice([True,False],p=[epsilon,1.0-epsilon])
    if rd: # 
        action = random.choice(actions) # 
    else: 
        action = actions[np.argmax(qline)] #  Q 
    return action
